1//! Motor abstraction for DRV8873 SPI driver with a TIM2 quadrature encoder.
2//!
3//! This module connects `Drv8873` and `Encoder<TIM2>`. The SPI bus is not owned here and must be
4//! passed in as `&mut SpiBus` so that multiple devices can share one bus safely.
5
6use crate::drivers::drv8873::{reg, Drv8873, Response, Status};
7use crate::hw::{Encoder, SpiBus};
8use stm32f7xx_hal::{pac, spi};
9
10/// Logical drive direction for the motor.
11#[derive(Copy, Clone, Debug)]
12pub enum Direction {
13    Forward,
14    Reverse,
15    Brake,
16    Coast,
17}

19/// High-level motor that combines a DRV8873 driver IC (SPI + CS) and a TIM2 32-bit encoder.
20pub struct SpiMotor<const P: char, const N: u8> {
21    drv: Drv8873<P, N>,
22    enc: Encoder<pac::TIM2>,
23    counts_per_rev: u32,
24}

151    /// Raw encoder position in ticks, centered around 0.
152    #[inline]
153    pub fn position_ticks(&self) -> i32 {
154        self.enc.position()
155    }
156
157    /// Encoder position converted to revolutions (float).
158    #[inline]
159    pub fn position_revs(&self) -> f32 {
160        self.enc.position() as f32 / self.counts_per_rev as f32
161    }
162
163    /// Reset the encoder position to zero.
164    #[inline]
165    pub fn zero(&mut self) {
166        self.enc.reset();
167    }

169    /// Expose the underlying encoder if needed for advanced usage.

171    pub fn encoder(&self) -> &Encoder<pac::TIM2> {

175    /// Mutable access to the encoder.
