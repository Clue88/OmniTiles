1//! Closed-loop controller for the linear lift actuator.
2//!
3//! This controller wraps a [`LiftMotor`] and provides a periodic `step()` function that computes a
4//! drive command using PID control and applies it to the motor.
5//!
6//! Typical usage pattern:
7//!
8//! ```no_run
9//! controller.set_target_height_mm(150.0);
10//!
11//! loop {
12//!     controller.step(dt_seconds);
13//!     delay.delay_ms(10_u32);
14//! }
15//! ```
16
17use crate::control::Pid;
18use crate::motors::LiftMotor;
19
20/// Operating mode of the lift controller.
21#[derive(Copy, Clone, Debug)]
22pub enum LiftMode {
23    /// Regular closed-loop control toward a height target.
24    PositionControl,
25
26    /// Motor is manually disabled (coast).
27    Disabled,
28}
29
30/// Controller state and configuration.
31pub struct LiftController<
32    const CS_P: char,
33    const CS_N: u8,
34    const IN1_P: char,
35    const IN1_N: u8,
36    const IN2_P: char,
37    const IN2_N: u8,
38    const SLP_P: char,
39    const SLP_N: u8,
40    const DIS_P: char,
41    const DIS_N: u8,
42> {
43    motor: LiftMotor<CS_P, CS_N, IN1_P, IN1_N, IN2_P, IN2_N, SLP_P, SLP_N, DIS_P, DIS_N>,
44    pid: Pid,
45    mode: LiftMode,
46
47    /// Commanded height (mm)
48    target_height_mm: f32,
49
50    /// Safety limits (mm)
51    min_height_mm: f32,
52    max_height_mm: f32,
53
54    /// Deadband in mm, "close enough" to target
55    on_target_tolerance_mm: f32,
56}
57
58impl<
59        const CS_P: char,
60        const CS_N: u8,
61        const IN1_P: char,
62        const IN1_N: u8,
63        const IN2_P: char,
64        const IN2_N: u8,
65        const SLP_P: char,
66        const SLP_N: u8,
67        const DIS_P: char,
68        const DIS_N: u8,
69    > LiftController<CS_P, CS_N, IN1_P, IN1_N, IN2_P, IN2_N, SLP_P, SLP_N, DIS_P, DIS_N>
70{
71    /// Create a new lift controller with PID gains and limits.
72    pub fn new(
73        motor: LiftMotor<CS_P, CS_N, IN1_P, IN1_N, IN2_P, IN2_N, SLP_P, SLP_N, DIS_P, DIS_N>,
74        pid: Pid,
75        min_height_mm: f32,
76        max_height_mm: f32,
77        on_target_tolerance_mm: f32,
78    ) -> Self {
79        Self {
80            motor,
81            pid,
82            mode: LiftMode::PositionControl,
83            target_height_mm: 0.0,
84            min_height_mm,
85            max_height_mm,
86            on_target_tolerance_mm,
87        }
88    }
89
90    /// Set a new target height (mm), automatically clamped to limits.
91    pub fn set_target_height_mm(&mut self, mm: f32) {
92        let clamped = mm.clamp(self.min_height_mm, self.max_height_mm);
93        self.target_height_mm = clamped;
94        self.pid.reset();
95    }
96
97    /// Directly disable the lift (coast).
98    pub fn disable(&mut self) {
99        self.mode = LiftMode::Disabled;
100        self.motor.inner_motor().coast();
101    }
102
103    /// Re-enable position control.
104    pub fn enable(&mut self) {
105        self.mode = LiftMode::PositionControl;
106    }
107
108    /// Current measured height (mm).
109    #[inline]
110    pub fn height_mm(&self) -> f32 {
111        self.motor.height_mm()
112    }
113
114    /// Returns true if the lift is within tolerance of its commanded height.
115    pub fn on_target(&self) -> bool {
116        let err = (self.height_mm() - self.target_height_mm).abs();
117        err <= self.on_target_tolerance_mm
118    }
119
120    /// Run one control step.
121    ///
122    /// `dt` â€” time delta in seconds since the previous call, e.g. 0.01 for 10ms.
123    pub fn step(&mut self, dt: f32) {
124        match self.mode {
125            LiftMode::Disabled => {
126                self.motor.inner_motor().coast();
127                return;
128            }
129
130            LiftMode::PositionControl => {
131                let current = self.motor.height_mm();
132                let target = self.target_height_mm;
133                let error = target - current;
134
135                // PID output
136                let mut u = self.pid.update(target, current, dt);
137
138                // Clamp to [-1.0, 1.0]
139                if u > 1.0 {
140                    u = 1.0;
141                }
142                if u < -1.0 {
143                    u = -1.0;
144                }
145
146                // Near-target deadband
147                if error.abs() <= self.on_target_tolerance_mm {
148                    self.motor.inner_motor().brake();
149                    return;
150                }
151
152                // Apply PID output
153                self.motor.apply_pid_output(u);
154            }
155        }
156    }
157}
