1//! Generic PID controller for closed-loop control.
2//!
3//! Works in `no_std` and does not allocate memory.
4
5/// PID controller with tunable gains and output clamping.
6pub struct Pid {
7    /// Proportional gain
8    kp: f32,
9    /// Integral gain
10    ki: f32,
11    /// Derivative gain
12    kd: f32,
13
14    /// Integrator state
15    integral: f32,
16    /// Last process variable (for derivative term)
17    prev_measurement: f32,
18
19    /// Output clamp
20    out_min: f32,
21    out_max: f32,
22
23    /// Integral anti-windup clamp
24    int_min: f32,
25    int_max: f32,
26
27    first_update: bool,
28}
29
30impl Pid {
31    /// Create a new PID controller.
32    ///
33    /// `kp`, `ki`, `kd` are the gain constants.
34    pub fn new(kp: f32, ki: f32, kd: f32) -> Self {
35        Self {
36            kp,
37            ki,
38            kd,
39
40            integral: 0.0,
41            prev_measurement: 0.0,
42
43            out_min: -1.0,
44            out_max: 1.0,
45
46            int_min: -1.0,
47            int_max: 1.0,
48
49            first_update: true,
50        }
51    }
52
53    /// Set output limits.
54    pub fn with_output_limits(mut self, min: f32, max: f32) -> Self {
55        self.out_min = min;
56        self.out_max = max;
57        self
58    }
59
60    /// Set integral limits for anti-windup.
61    pub fn with_integral_limits(mut self, min: f32, max: f32) -> Self {
62        self.int_min = min;
63        self.int_max = max;
64        self
65    }
66
67    /// Reset integrator + derivative history.
68    pub fn reset(&mut self) {
69        self.integral = 0.0;
70        self.prev_measurement = 0.0;
71        self.first_update = true;
72    }
73
74    /// Update the controller.
75    ///
76    /// `setpoint` — desired value  
77    /// `measurement` — current value  
78    /// `dt` — timestep in seconds (e.g. 0.02 for 50 Hz control loop)
79    ///
80    /// Returns a normalized command in [`out_min`, `out_max`] which can be mapped to motor drive.
81    pub fn update(&mut self, setpoint: f32, measurement: f32, dt: f32) -> f32 {
82        let error = setpoint - measurement;
83
84        // ----- P term -----
85        let p = self.kp * error;
86
87        // ----- I term -----
88        self.integral += error * dt * self.ki;
89
90        // Anti-windup clamp
91        if self.integral > self.int_max {
92            self.integral = self.int_max;
93        }
94        if self.integral < self.int_min {
95            self.integral = self.int_min;
96        }
97
98        let i = self.integral;
99
100        // ----- D term (on measurement to reduce noise sensitivity) -----
101        let d = if self.first_update {
102            self.first_update = false;
103            0.0
104        } else {
105            let dv = self.prev_measurement - measurement;
106            self.kd * (dv / dt)
107        };
108        self.prev_measurement = measurement;
109
110        // ----- Output clamp -----
111        let mut out = p + i + d;
112        if out > self.out_max {
113            out = self.out_max;
114        }
115        if out < self.out_min {
116            out = self.out_min;
117        }
118
119        out
120    }
121}
