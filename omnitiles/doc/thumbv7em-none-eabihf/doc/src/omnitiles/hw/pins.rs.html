1//! Central definition of all STM32F777 pin assignments used by OmniTiles.
2
3use stm32f7xx_hal::{
4    gpio::{gpioa, gpiob, gpiod, gpioe, Alternate, Output, PushPull},
5    pac,
6    prelude::*,
7};
8
9/// All board pins. Construct this once at startup using:
10///
11/// ```rust
12/// let pins = BoardPins::new(dp.GPIOA, dp.GPIOB, dp.GPIOD, dp.GPIOE);
13/// ```
14pub struct BoardPins {
15    pub leds: LedPins,
16    pub usart1: Usart1Pins,
17    pub spi4: Spi4Pins,
18    pub drv8873: DrvPins,
19    pub encoder: EncoderPins,
20    pub can1: Can1Pins,
21    pub can2: Can2Pins,
22}
23
24pub struct LedPins {
25    pub red: gpiod::PD8<Output<PushPull>>,
26    pub yellow: gpiod::PD9<Output<PushPull>>,
27    pub green: gpiod::PD10<Output<PushPull>>,
28}
29
30pub struct Usart1Pins {
31    pub tx: gpioa::PA9<Alternate<7>>,
32    pub rx: gpioa::PA10<Alternate<7>>,
33}
34
35/// SPI4 SCK/MISO/MOSI
36pub struct Spi4Pins {
37    pub sck: gpioe::PE12<Alternate<5>>,
38    pub miso: gpioe::PE13<Alternate<5>>,
39    pub mosi: gpioe::PE14<Alternate<5>>,
40}
41
42/// Chip-select and digital I/O for DRV8873 drivers
43pub struct DrvPins {
44    pub m1_cs: gpioe::PE4<Output<PushPull>>,
45    pub m2_cs: gpioe::PE11<Output<PushPull>>,
46}
47
48/// TIM2/TIM3 Quadrature Encoder Inputs
49pub struct EncoderPins {
50    pub tim2_ch1: gpioa::PA0<Alternate<1>>,
51    pub tim2_ch2: gpioa::PA1<Alternate<1>>,
52
53    pub tim3_ch1: gpioa::PA6<Alternate<2>>,
54    pub tim3_ch2: gpioa::PA7<Alternate<2>>,
55}
56
57/// CAN1 bus pins
58pub struct Can1Pins {
59    pub tx: gpioa::PA12<Alternate<9>>,
60    pub rx: gpioa::PA11<Alternate<9>>,
61}
62
63/// CAN2 bus pins
64pub struct Can2Pins {
65    pub tx: gpiob::PB13<Alternate<9>>,
66    pub rx: gpiob::PB12<Alternate<9>>,
67}
68
69impl BoardPins {
70    /// Create all named pins from raw GPIO peripherals.
71    pub fn new(gpioa: pac::GPIOA, gpiob: pac::GPIOB, gpiod: pac::GPIOD, gpioe: pac::GPIOE) -> Self {
72        let gpioa = gpioa.split();
73        let gpiob = gpiob.split();
74        let gpiod = gpiod.split();
75        let gpioe = gpioe.split();
76
77        Self {
78            leds: LedPins {
79                red: gpiod.pd8.into_push_pull_output(),
80                yellow: gpiod.pd9.into_push_pull_output(),
81                green: gpiod.pd10.into_push_pull_output(),
82            },
83
84            usart1: Usart1Pins {
85                tx: gpioa.pa9.into_alternate::<7>(),
86                rx: gpioa.pa10.into_alternate::<7>(),
87            },
88
89            spi4: Spi4Pins {
90                sck: gpioe.pe12.into_alternate::<5>(),
91                miso: gpioe.pe13.into_alternate::<5>(),
92                mosi: gpioe.pe14.into_alternate::<5>(),
93            },
94
95            drv8873: DrvPins {
96                m1_cs: gpioe.pe4.into_push_pull_output(),
97                m2_cs: gpioe.pe11.into_push_pull_output(),
98            },
99
100            encoder: EncoderPins {
101                tim2_ch1: gpioa.pa0.into_alternate::<1>(),
102                tim2_ch2: gpioa.pa1.into_alternate::<1>(),
103                tim3_ch1: gpioa.pa6.into_alternate::<2>(),
104                tim3_ch2: gpioa.pa7.into_alternate::<2>(),
105            },
106
107            can1: Can1Pins {
108                tx: gpioa.pa12.into_alternate::<9>(),
109                rx: gpioa.pa11.into_alternate::<9>(),
110            },
111
112            can2: Can2Pins {
113                tx: gpiob.pb13.into_alternate::<9>(),
114                rx: gpiob.pb12.into_alternate::<9>(),
115            },
116        }
117    }
118}
